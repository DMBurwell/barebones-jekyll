###
###  LAYOUT INHERITANCE
###
###  pages built by adding elements 
###  to other page layouts 
###

      default.html layout 
      │
      └── format-text.html layout 
          │
          └── add-table.html layout
              # see liquid table below



      ###
      ### default.html
      ###
      ### include statements add page 
      ### components from _includes folder
      ###
      
      <!DOCTYPE html>
      <html lang="en">
      {% include head.html %}
      
        <body>                                
          {% include header.html %}        
          {{ content }}                      
          {% include footer.html %}           
        </body>  
        
      </html>     
      
      
      
      ###
      ### format-text.html
      ###
      
      ---
      layout: default
      ---

      <div class="pretty-text">

      <h1> {{ page.title }} </h1>

      {{ content }}

      </div>
      
      
      ###
      ### add-table.html
      ###
      
      ---
      layout: format-text
      ---

      {{ content }}

      #  liquid table starts here
      




###
###  LIQUID TABLE ADDED USING YAML DATA
###
###  liquid is a simple markup language that allows
###  web designers to separate page layouts from
###  page content without databases
###
###  {%  function in liquid  %}
###  {{  variable in liquid  }}
###

    ###
    ###  YAML HEADER FOR PAGE
    ###
    
    ---
    layout: add-table
    title: 'amiright?'
    reynolds:
      strengths:
